\textbf{code:}
\begin{lstlisting}

# euler's method
x = 0.0
z = 1.0
h = 0.1
steps = 5

for k in range(steps):
    f = x - 2*z
    f_times_h = f * h
    z_next = z + f_times_h
    
    print(f"{k:<5} {x:<8.1f} {z:<8.4f} {f:<12.4f} #table variables
           {f_times_h:<10.4f} {z_next:<8.4f}")
    
    x = x + h
    z = z_next
\end{lstlisting}

# RK2 midpoint method
x = 0.0
z = 1.0
h = 0.1
steps = 5

for k in range(steps):
    k1 = x - 2*z
    k2 = (x + h/2) - 2*(z + k1*h/2)
    z_next = z + k2*h
    
    print(f"{k:<5} {x:<8.1f} {z:<10.4f} {k1:<10.4f} #table variables
           {k2:<10.4f} {z_next:<10.4f}") 
    
    x = x + h
    z = z_next
\end{lstlisting}
